CREATE TABLE `studentattendance`.`teachers` ( `T_id` INT(5) NOT NULL AUTO_INCREMENT , `Name` VARCHAR(30) NOT NULL , `Email` VARCHAR(30) NOT NULL , `Phone` VARCHAR(15) NOT NULL , `Username` VARCHAR(20) NOT NULL , `Password` VARCHAR(20) NOT NULL , `Created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , `Status` BOOLEAN NOT NULL , PRIMARY KEY (`T_id`)) ENGINE = InnoDB;
